# Docker - How to Move Docker to Another Drive on Windows
# exit from docker desktop app

wsl -l -v
wsl --shotdown
mkdir E:\backup

wsl --export docker-desktop-data E:\backup\dockerdesktop.tar
wsl --unregister docker-desktop-data
wsl --import docker-desktop-data E:\docker\ E:\backup\dockerdesktop.tar
wsl -l -v
